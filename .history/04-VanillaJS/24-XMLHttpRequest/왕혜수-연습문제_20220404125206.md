              tr.appendChild(td2);
              tr.appendChild(td3);

            });

          .catch((error) => {
            console.error(error);
            console.error(error.response.status);
            console.error(error.response.statusText);
            console.error(error.response.data);
            alert(error.response.statusText);
          })
          .finally(() => {
            //성공,실패 여부에 상관 없이 마지막에 실행되는 함수
            loading.classList.remove('active');
          });
      });
    </script>
