## 문제2 (시간내 풀지 못하여 필사만 하였습니다)

```javascript
function Rectangle() {
  this._width = null;
  this._height = null;
}

Rectangle.prototype = {
  get width() {
    return this._width;
  },
  set width(param) {
    this._width = param;
  },
  get height() {
    return this._height;
  },
  set height(param) {
    this._height = param;
  },
  getAround: function () {
    return this.width * 2 + this.height * 2;
  },
  getArea: function () {
    return this.width * this.height;
  },
};

const rect = new Rectangle();
rect.width = 10;
rect.height = 5;

console.log(
  '둘레 길이는 %d이고 넚이는 %d입니다',
  rect.getAround(),
  rect.getArea()
);
```
