## 문제1

```javascript
function printStar(max){ 
    for (let i =0; i < max; i++){
        let str = '';
        for (let j = 0; j < i + 1; j++) {
        str += '*';
        }console.log(str);
    }
}
printStar(5);
```

## 문제2


```javascript
function printRevStar(max){
    for (let i =0; i < max; i++){
        let str = '';
        for (let j = 0; j < max - i; j++) {
        str += '*';
        }console.log(str);
    }
}
printRevStar(5);
```
